#include using namespace std;
int lowerBound(vector arr, int n, int x) {
int low = 0, high = n - 1;
int ans = n;
while (low <= high) {
int mid = (low + high) / 2;
// maybe an answer
if (arr[mid] >= x) {
ans = mid;
//look for smaller index on the left
high = mid - 1;
} else {
low = mid + 1; // look on the right }
} return ans;
} int main()
{ vector arr = {3, 5, 8, 15, 19};
//index = 0, 1, 2, 3, 4
int n = 5, x = 2;
int ind = lowerBound(arr, n, x);
cout << "The lower bound is the index: " << ind << "\n";
return 0;
}
